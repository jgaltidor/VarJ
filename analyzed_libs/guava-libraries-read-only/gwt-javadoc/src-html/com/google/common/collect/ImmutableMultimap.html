032    /**
033     * An immutable {@link Multimap}. Does not permit null keys or values.
034     *
035     * <p>Unlike {@link Multimaps#unmodifiableMultimap(Multimap)}, which is
036     * a <i>view</i> of a separate multimap which can still change, an instance of
037     * {@code ImmutableMultimap} contains its own data and will <i>never</i>
038     * change. {@code ImmutableMultimap} is convenient for
039     * {@code public static final} multimaps ("constant multimaps") and also lets
040     * you easily make a "defensive copy" of a multimap provided to your class by
041     * a caller.
042     *
043     * <p><b>Note</b>: Although this class is not final, it cannot be subclassed as
044     * it has no public or protected constructors. Thus, instances of this class
045     * are guaranteed to be immutable.
046     *
047     * @author Jared Levy
048     * @since 2 (imported from Google Collections Library)
049     */
050    @GwtCompatible(emulated = true)
051    public abstract class ImmutableMultimap<K, V>
052        implements Multimap<K, V>, Serializable {
053    
054      /** Returns an empty multimap. */
055      public static <K, V> ImmutableMultimap<K, V> of() {
056        return ImmutableListMultimap.of();
057      }

061       */

063        return ImmutableListMultimap.of(k1, v1);
064      }

070        return ImmutableListMultimap.of(k1, v1, k2, v2);

107      /**
108       * Multimap for {@link ImmutableMultimap.Builder} that maintains key and
109       * value orderings, allows duplicate values, and performs better than
110       * {@link LinkedListMultimap}.
111       */
112      private static class BuilderMultimap<K, V> extends AbstractMultimap<K, V> {
113        BuilderMultimap() {
114          super(new LinkedHashMap<K, Collection<V>>());
115        }
116        @Override Collection<V> createCollection() {
117          return Lists.newArrayList();
118        }
119        private static final long serialVersionUID = 0;
120      }

122      /**
123       * A builder for creating immutable multimap instances, especially
124       * {@code public static final} multimaps ("constant multimaps"). Example:
125       * <pre>   {@code
126       *
127       *   static final Multimap<String, Integer> STRING_TO_INTEGER_MULTIMAP =
128       *       new ImmutableMultimap.Builder<String, Integer>()
129       *           .put("one", 1)
130       *           .putAll("several", 1, 2, 3)
131       *           .putAll("many", 1, 2, 3, 4, 5)
132       *           .build();}</pre>
133       *
134       * Builder instances can be reused; it is safe to call {@link #build} multiple
135       * times to build multiple multimaps in series. Each multimap contains the
136       * key-value mappings in the previously created multimaps.
137       *
138       * @since 2 (imported from Google Collections Library)
139       */
140      public static class Builder<K, V> {
141        private final Multimap<K, V> builderMultimap = new BuilderMultimap<K, V>();
142    
143        /**
144         * Creates a new builder. The returned builder is equivalent to the builder
145         * generated by {@link ImmutableMultimap#builder}.
146         */
147        public Builder() {}
148    
149        /**
150         * Adds a key-value mapping to the built multimap.
151         */
152        public Builder<K, V> put(K key, V value) {
153          builderMultimap.put(checkNotNull(key), checkNotNull(value));
154          return this;
155        }
156    
157        /**
158         * Stores a collection of values with the same key in the built multimap.
159         *
160         * @throws NullPointerException if {@code key}, {@code values}, or any
161         *     element in {@code values} is null. The builder is left in an invalid
162         *     state.
163         */
164        public Builder<K, V> putAll(K key, Iterable<? extends V> values) {
165          Collection<V> valueList = builderMultimap.get(checkNotNull(key));
166          for (V value : values) {
167            valueList.add(checkNotNull(value));
168          }
169          return this;
170        }
171    
172        /**
173         * Stores an array of values with the same key in the built multimap.
174         *
175         * @throws NullPointerException if the key or any value is null. The builder
176         *     is left in an invalid state.
177         */
178        public Builder<K, V> putAll(K key, V... values) {
179          return putAll(key, Arrays.asList(values));
180        }
181    
182        /**
183         * Stores another multimap's entries in the built multimap. The generated
184         * multimap's key and value orderings correspond to the iteration ordering
185         * of the {@code multimap.asMap()} view, with new keys and values following
186         * any existing keys and values.
187         *
188         * @throws NullPointerException if any key or value in {@code multimap} is
189         *     null. The builder is left in an invalid state.
190         */
191        public Builder<K, V> putAll(Multimap<? extends K, ? extends V> multimap) {
192          for (Map.Entry<? extends K, ? extends Collection<? extends V>> entry
193              : multimap.asMap().entrySet()) {
194            putAll(entry.getKey(), entry.getValue());
195          }
196          return this;
197        }
198    
199        /**
200         * Returns a newly-created immutable multimap.
201         */
202        public ImmutableMultimap<K, V> build() {
203          return copyOf(builderMultimap);
204        }
205      }
206    
207      /**
208       * Returns an immutable multimap containing the same mappings as {@code
209       * multimap}. The generated multimap's key and value orderings correspond to
210       * the iteration ordering of the {@code multimap.asMap()} view.
211       *
212       * <p>Despite the method name, this method attempts to avoid actually copying
213       * the data when it is safe to do so. The exact circumstances under which a
214       * copy will or will not be performed are undocumented and subject to change.
215       *
216       * @throws NullPointerException if any key or value in {@code multimap} is
217       *         null
218       */
219      public static <K, V> ImmutableMultimap<K, V> copyOf(
220          Multimap<? extends K, ? extends V> multimap) {
221        if (multimap instanceof ImmutableMultimap) {
222          @SuppressWarnings("unchecked") // safe since multimap is not writable
223          ImmutableMultimap<K, V> kvMultimap
224              = (ImmutableMultimap<K, V>) multimap;
225          if (!kvMultimap.isPartialView()) {
226            return kvMultimap;
227          }
228        }
229        return ImmutableListMultimap.copyOf(multimap);
230      }

335        for (Collection<V> valueCollection : map.values()) {
336          if (valueCollection.contains(value)) {

338          }

387      private transient ImmutableCollection<Map.Entry<K, V>> entries;
388    
389      /**
390       * Returns an immutable collection of all key-value pairs in the multimap. Its
391       * iterator traverses the values for the first key, the values for the second
392       * key, and so on.
393       */
394      public ImmutableCollection<Map.Entry<K, V>> entries() {
395        ImmutableCollection<Map.Entry<K, V>> result = entries;
396        return (result == null)
397            ? (entries = new EntryCollection<K, V>(this)) : result;
398      }
399    
400      private static class EntryCollection<K, V>
401          extends ImmutableCollection<Map.Entry<K, V>> {
402        final ImmutableMultimap<K, V> multimap;
403    
404        EntryCollection(ImmutableMultimap<K, V> multimap) {
405          this.multimap = multimap;
406        }
407    
408        @Override public UnmodifiableIterator<Map.Entry<K, V>> iterator() {
409          final Iterator<? extends Map.Entry<K, ? extends ImmutableCollection<V>>>
410              mapIterator = this.multimap.map.entrySet().iterator();
411    
412          return new UnmodifiableIterator<Map.Entry<K, V>>() {
413            K key;
414            Iterator<V> valueIterator;
415    
416            public boolean hasNext() {
417              return (key != null && valueIterator.hasNext())
418                  || mapIterator.hasNext();
419            }
420    
421            public Map.Entry<K, V> next() {
422              if (key == null || !valueIterator.hasNext()) {
423                Map.Entry<K, ? extends ImmutableCollection<V>> entry
424                    = mapIterator.next();
425                key = entry.getKey();
426                valueIterator = entry.getValue().iterator();
427              }
428              return Maps.immutableEntry(key, valueIterator.next());
429            }
430          };
431        }
432    
433        @Override boolean isPartialView() {
434          return multimap.isPartialView();
435        }
436    
437        public int size() {
438          return multimap.size();
439        }
440    
441        @Override public boolean contains(Object object) {
442          if (object instanceof Map.Entry) {
443            Map.Entry<?, ?> entry = (Map.Entry<?, ?>) object;
444            return multimap.containsEntry(entry.getKey(), entry.getValue());
445          }
446          return false;
447        }
448    
449        private static final long serialVersionUID = 0;
450      }

452      private transient ImmutableMultiset<K> keys;
453    
454      /**
455       * Returns a collection, which may contain duplicates, of all keys. The number
456       * of times a key appears in the returned multiset equals the number of
457       * mappings the key has in the multimap. Duplicate keys appear consecutively
458       * in the multiset's iteration order.
459       */
460      public ImmutableMultiset<K> keys() {
461        ImmutableMultiset<K> result = keys;
462        return (result == null) ? (keys = createKeys()) : result;
463      }
464    
465      private ImmutableMultiset<K> createKeys() {
466        ImmutableMultiset.Builder<K> builder = ImmutableMultiset.builder();
467        for (Map.Entry<K, ? extends ImmutableCollection<V>> entry
468            : map.entrySet()) {
469          builder.addCopies(entry.getKey(), entry.getValue().size());
470        }
471        return builder.build();
472      }

476      /**
477       * Returns an immutable collection of the values in this multimap. Its
478       * iterator traverses the values for the first key, the values for the second
479       * key, and so on.
480       */
481      public ImmutableCollection<V> values() {
482        ImmutableCollection<V> result = values;
483        return (result == null) ? (values = new Values<V>(this)) : result;
484      }
485    
486      private static class Values<V> extends ImmutableCollection<V> {
487        final ImmutableMultimap<?, V> multimap;
488    
489        Values(ImmutableMultimap<?, V> multimap) {
490          this.multimap = multimap;
491        }
492    
493        @Override public UnmodifiableIterator<V> iterator() {
494          final Iterator<? extends Map.Entry<?, V>> entryIterator
495              = multimap.entries().iterator();
496          return new UnmodifiableIterator<V>() {
497            public boolean hasNext() {
498              return entryIterator.hasNext();
499            }
500            public V next() {
501              return entryIterator.next().getValue();
502            }
503          };
504        }
505    
506        public int size() {
507          return multimap.size();
508        }
509    
510        @Override boolean isPartialView() {
511          return true;
512        }
513    
514        private static final long serialVersionUID = 0;
515      }
