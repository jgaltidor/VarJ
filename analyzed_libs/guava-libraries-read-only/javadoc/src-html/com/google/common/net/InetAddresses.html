035    /**
036     * Static utility methods pertaining to {@link InetAddress} instances.
037     *
038     * <p><b>Important note:</b> Unlike {@code InetAddress.getByName()}, the
039     * methods of this class never cause DNS services to be accessed. For
040     * this reason, you should prefer these methods as much as possible over
041     * their JDK equivalents whenever you are expecting to handle only
042     * IP address string literals -- there is no blocking DNS penalty for a
043     * malformed string.
044     *
045     * <p>This class hooks into the {@code sun.net.util.IPAddressUtil} class
046     * to make use of the {@code textToNumericFormatV4} and
047     * {@code textToNumericFormatV6} methods directly as a means to avoid
048     * accidentally traversing all nameservices (it can be vitally important
049     * to avoid, say, blocking on DNS at times).
050     *
051     * <p>When dealing with {@link Inet4Address} and {@link Inet6Address}
052     * objects as byte arrays (vis. {@code InetAddress.getAddress()}) they
053     * are 4 and 16 bytes in length, respectively, and represent the address
054     * in network byte order.
055     *
056     * <p>Examples of IP addresses and their byte representations:
057     * <ul>
058     * <li>The IPv4 loopback address, {@code "127.0.0.1"}.<br/>
059     *     {@code 7f 00 00 01}
060     *
061     * <li>The IPv6 loopback address, {@code "::1"}.<br/>
062     *     {@code 00 00 00 00 00 00 00 00 00 00 00 00 00 00 00 01}
063     *
064     * <li>From the IPv6 reserved documentation prefix ({@code 2001:db8::/32}),
065     *     {@code "2001:db8::1"}.<br/>
066     *     {@code 20 01 0d b8 00 00 00 00 00 00 00 00 00 00 00 01}
067     *
068     * <li>An IPv6 "IPv4 compatible" (or "compat") address,
069     *     {@code "::192.168.0.1"}.<br/>
070     *     {@code 00 00 00 00 00 00 00 00 00 00 00 00 c0 a8 00 01}
071     *
072     * <li>An IPv6 "IPv4 mapped" address, {@code "::ffff:192.168.0.1"}.<br/>
073     *     {@code 00 00 00 00 00 00 00 00 00 00 ff ff c0 a8 00 01}
074     * </ul>
075     *
076     * <p>A few notes about IPv6 "IPv4 mapped" addresses and their observed
077     * use in Java.
078     * <br><br>
079     * "IPv4 mapped" addresses were originally a representation of IPv4
080     * addresses for use on an IPv6 socket that could receive both IPv4
081     * and IPv6 connections (by disabling the {@code IPV6_V6ONLY} socket
082     * option on an IPv6 socket).  Yes, it's confusing.  Nevertheless,
083     * these "mapped" addresses were never supposed to be seen on the
084     * wire.  That assumption was dropped, some say mistakenly, in later
085     * RFCs with the apparent aim of making IPv4-to-IPv6 transition simpler.
086     *
087     * <p>Technically one <i>can</i> create a 128bit IPv6 address with the wire
088     * format of a "mapped" address, as shown above, and transmit it in an
089     * IPv6 packet header.  However, Java's InetAddress creation methods
090     * appear to adhere doggedly to the original intent of the "mapped"
091     * address: all "mapped" addresses return {@link Inet4Address} objects.
092     *
093     * <p>For added safety, it is common for IPv6 network operators to filter
094     * all packets where either the source or destination address appears to
095     * be a "compat" or "mapped" address.  Filtering suggestions usually
096     * recommend discarding any packets with source or destination addresses
097     * in the invalid range {@code ::/3}, which includes both of these bizarre
098     * address formats.  For more information on "bogons", including lists
099     * of IPv6 bogon space, see:
100     *
101     * <ul>
102     * <li><a target="_parent"
103     *        href="http://en.wikipedia.org/wiki/Bogon_filtering"
104     *       >http://en.wikipedia.org/wiki/Bogon_filtering</a>
105     * <li><a target="_parent"
106     *        href="http://www.cymru.com/Bogons/ipv6.txt"
107     *       >http://www.cymru.com/Bogons/ipv6.txt</a>
108     * <li><a target="_parent"
109     *        href="http://www.cymru.com/Bogons/v6bogon.html"
110     *       >http://www.cymru.com/Bogons/v6bogon.html</a>
111     * <li><a target="_parent"
112     *        href="http://www.space.net/~gert/RIPE/ipv6-filters.html"
113     *       >http://www.space.net/~gert/RIPE/ipv6-filters.html</a>
114     * </ul>
115     *
116     * @author Erik Kline
117     * @since 5
118     */

130      /**
131       * Returns an {@link Inet4Address}, given a byte array representation
132       * of the IPv4 address.
133       *
134       * @param bytes byte array representing an IPv4 address (should be
135       *              of length 4).
136       * @return {@link Inet4Address} corresponding to the supplied byte
137       *         array.
138       * @throws IllegalArgumentException if a valid {@link Inet4Address}
139       *         can not be created.
140       */
141      private static Inet4Address getInet4Address(byte[] bytes) {
142        Preconditions.checkArgument(bytes.length == 4,
143            "Byte array has invalid length for an IPv4 address: %s != 4.",
144            bytes.length);
145    
146        try {
147          InetAddress ipv4 = InetAddress.getByAddress(bytes);
148          if (!(ipv4 instanceof Inet4Address)) {
149            throw new UnknownHostException(
150                String.format("'%s' is not an IPv4 address.",
151                              ipv4.getHostAddress()));
152          }
153    
154          return (Inet4Address) ipv4;
155        } catch (UnknownHostException e) {
156    
157          /*
158           * This really shouldn't happen in practice since all our byte
159           * sequences should be valid IP addresses.
160           *
161           * However {@link InetAddress#getByAddress} is documented as
162           * potentially throwing this "if IP address is of illegal length".
163           *
164           * This is mapped to IllegalArgumentException since, presumably,
165           * the argument triggered some bizarre processing bug.
166           */
167          throw new IllegalArgumentException(
168              String.format("Host address '%s' is not a valid IPv4 address.",
169                            Arrays.toString(bytes)),
170              e);
171        }
172      }

174      /**
175       * Returns the {@link InetAddress} having the given string
176       * representation.
177       *
178       * <p>This deliberately avoids all nameservice lookups (e.g. no DNS).
179       *
180       * @param ipString {@code String} containing an IPv4 or IPv6 string literal,
181       *                 e.g. {@code "192.168.0.1"} or {@code "2001:db8::1"}
182       * @return {@link InetAddress} representing the argument
183       * @throws IllegalArgumentException if the argument is not a valid
184       *         IP string literal
185       */
186      public static InetAddress forString(String ipString) {
187        byte[] addr = textToNumericFormatV4(ipString);
188        if (addr == null) {
189          // Scanning for IPv4 string literal failed; try IPv6.
190          addr = textToNumericFormatV6(ipString);
191        }
192    
193        // The argument was malformed, i.e. not an IP string literal.
194        if (addr == null) {
195          throw new IllegalArgumentException(
196              String.format("'%s' is not an IP string literal.", ipString));
197        }
198    
199        try {
200          return InetAddress.getByAddress(addr);
201        } catch (UnknownHostException e) {
202    
203          /*
204           * This really shouldn't happen in practice since all our byte
205           * sequences should be valid IP addresses.
206           *
207           * However {@link InetAddress#getByAddress} is documented as
208           * potentially throwing this "if IP address is of illegal length".
209           *
210           * This is mapped to IllegalArgumentException since, presumably,
211           * the argument triggered some processing bug in either
212           * {@link IPAddressUtil#textToNumericFormatV4} or
213           * {@link IPAddressUtil#textToNumericFormatV6}.
214           */
215          throw new IllegalArgumentException(
216              String.format("'%s' is extremely broken.", ipString), e);
217        }
218      }

221       * Returns {@code true} if the supplied string is a valid IP string
222       * literal, {@code false} otherwise.
223       *
224       * @param ipString {@code String} to evaluated as an IP string literal
225       * @return {@code true} if the argument is a valid IP string literal
226       */
227      public static boolean isInetAddress(String ipString) {
228        try {
229          forString(ipString);
230          return true;
231        } catch (IllegalArgumentException e) {
232          return false;
233        }
234      }

236      private static byte[] textToNumericFormatV4(String ipString) {
237    
238        boolean isIpv6 = false;
239    
240        // handle IPv6 forms of IPv4 addresses
241        // TODO: use Ascii.toUpperCase() when available
242        if (ipString.toUpperCase(Locale.US).startsWith("::FFFF:")) {
243          ipString = ipString.substring(7);
244        } else if (ipString.startsWith("::")) {
245          ipString = ipString.substring(2);
246          isIpv6 = true;
247        }
248    
249        String[] address = ipString.split("\\.");
250        if (address.length != IPV4_PART_COUNT) {
251          return null;
252        }
253        try {
254          byte[] bytes = new byte[IPV4_PART_COUNT];
255          for (int i = 0; i < bytes.length; i++) {
256            int piece = Integer.parseInt(address[i]);
257            if (piece < 0 || piece > 255) {
258              return null;
259            }
260    
261            // No leading zeroes are allowed.  See
262            // http://tools.ietf.org/html/draft-main-ipaddr-text-rep-00
263            // section 2.1 for discussion.
264    
265            if (address[i].startsWith("0") && address[i].length() != 1) {
266              return null;
267            }
268            bytes[i] = (byte) piece;
269          }
270    
271          if (isIpv6) { // prepend with zeroes;
272            byte[] data = new byte[2 * IPV6_PART_COUNT]; // Java initializes arrays to zero
273            System.arraycopy(bytes, 0, data, 12, IPV4_PART_COUNT);
274            return data;
275          } else {
276            return bytes;
277          }
278        } catch (NumberFormatException ex) {
279          return null;
280        }
281      }
282    
283      private static byte[] textToNumericFormatV6(String ipString) {
284        if (!ipString.contains(":")) {
285          return null;
286        }
287        if (ipString.contains(":::")) {
288          return null;
289        }
290    
291        if (ipString.contains(".")) {
292          ipString = convertDottedQuadToHex(ipString);
293          if (ipString == null) {
294            return null;
295          }
296        }
297    
298        ipString = padIpString(ipString);
299        try {
300          String[] address = ipString.split(":", IPV6_PART_COUNT);
301          if (address.length != IPV6_PART_COUNT) {
302            return null;
303          }
304          byte[] bytes = new byte[2 * IPV6_PART_COUNT];
305          for (int i = 0; i < IPV6_PART_COUNT; i++) {
306            int piece = address[i].equals("") ? 0 : Integer.parseInt(address[i], 16);
307            bytes[2 * i] = (byte) ((piece & 0xFF00) >>> 8);
308            bytes[2 * i + 1] = (byte) (piece & 0xFF);
309          }
310          return bytes;
311        } catch (NumberFormatException ex) {
312          return null;
313        }
314      }

316      // Fill in any omitted colons
317      private static String padIpString(String ipString) {
318        if (ipString.contains("::")) {
319          int count = numberOfColons(ipString);
320          StringBuilder buffer = new StringBuilder("::");
321          for (int i = 0; i + count < 7; i++) {
322            buffer.append(":");
323          }
324          ipString = ipString.replace("::", buffer);
325        }
326        return ipString;
327      }
328    
329      private static int numberOfColons(String s) {
330        int count = 0;
331        for (int i = 0; i < s.length(); i++) {
332          if (s.charAt(i) == ':') {
333            count++;
334          }
335        }
336        return count;
337      }
338    
339      private static String convertDottedQuadToHex(String ipString) {
340        int lastColon = ipString.lastIndexOf(':');
341        String initialPart = ipString.substring(0, lastColon + 1);
342        String dottedQuad = ipString.substring(lastColon + 1);
343        byte[] quad = textToNumericFormatV4(dottedQuad);
344        if (quad == null) {
345          return null;
346        }
347        String penultimate = Integer.toHexString(((quad[0] & 0xff) << 8) | (quad[1] & 0xff));
348        String ultimate = Integer.toHexString(((quad[2] & 0xff) << 8) | (quad[3] & 0xff));
349        return initialPart + penultimate + ":" + ultimate;
350      }

352      /**
353       * Returns the string representation of an {@link InetAddress} suitable
354       * for inclusion in a URI.
355       *
356       * <p>For IPv4 addresses, this is identical to
357       * {@link InetAddress#getHostAddress()}, but for IPv6 addresses it
358       * surrounds this text with square brackets; for example
359       * {@code "[2001:db8::1]"}.
360       *
361       * <p>Per section 3.2.2 of
362       * <a target="_parent"
363       *    href="http://tools.ietf.org/html/rfc3986#section-3.2.2"
364       *  >http://tools.ietf.org/html/rfc3986</a>,
365       * a URI containing an IPv6 string literal is of the form
366       * {@code "http://[2001:db8::1]:8888/index.html"}.
367       *
368       * <p>Use of either {@link InetAddress#getHostAddress()} or this
369       * method is recommended over {@link InetAddress#toString()} when an
370       * IP address string literal is desired.  This is because
371       * {@link InetAddress#toString()} prints the hostname and the IP
372       * address string joined by a "/".
373       *
374       * @param ip {@link InetAddress} to be converted to URI string literal
375       * @return {@code String} containing URI-safe string literal
376       */
377      public static String toUriString(InetAddress ip) {
378        if (ip instanceof Inet6Address) {
379          return "[" + ip.getHostAddress() + "]";
380        }
381        return ip.getHostAddress();
382      }

384      /**
385       * Returns an InetAddress representing the literal IPv4 or IPv6 host
386       * portion of a URL, encoded in the format specified by RFC 3986 section 3.2.2.
387       *
388       * <p>This function is similar to {@link InetAddresses#forString(String)},
389       * however, it requires that IPv6 addresses are surrounded by square brackets.
390       *
391       * <p>This function is the inverse of
392       * {@link InetAddresses#toUriString(java.net.InetAddress)}.
393       *
394       * @param hostAddr A RFC 3986 section 3.2.2 encoded IPv4 or IPv6 address
395       * @return an InetAddress representing the address in {@code hostAddr}
396       * @throws IllegalArgumentException if {@code hostAddr} is not a valid
397       *     IPv4 address, or IPv6 address surrounded by square brackets
398       */
399      public static InetAddress forUriString(String hostAddr) {
400        Preconditions.checkNotNull(hostAddr);
401        Preconditions.checkArgument(hostAddr.length() > 0, "host string is empty");
402        InetAddress retval = null;
403    
404        // IPv4 address?
405        try {
406          retval = forString(hostAddr);
407          if (retval instanceof Inet4Address) {
408            return retval;
409          }
410        } catch (IllegalArgumentException e) {
411          // Not a valid IP address, fall through.
412        }
413    
414        // IPv6 address
415        if (!(hostAddr.startsWith("[") && hostAddr.endsWith("]"))) {
416          throw new IllegalArgumentException("Not a valid address: \"" + hostAddr + '"');
417        }
418    
419        retval = forString(hostAddr.substring(1, hostAddr.length() - 1));
420        if (retval instanceof Inet6Address) {
421          return retval;
422        }
423    
424        throw new IllegalArgumentException("Not a valid address: \"" + hostAddr + '"');
425      }

439          return false;

443      /**
444       * Evaluates whether the argument is an IPv6 "compat" address.
445       *
446       * <p>An "IPv4 compatible", or "compat", address is one with 96 leading
447       * bits of zero, with the remaining 32 bits interpreted as an
448       * IPv4 address.  These are conventionally represented in string
449       * literals as {@code "::192.168.0.1"}, though {@code "::c0a8:1"} is
450       * also considered an IPv4 compatible address (and equivalent to
451       * {@code "::192.168.0.1"}).
452       *
453       * <p>For more on IPv4 compatible addresses see section 2.5.5.1 of
454       * <a target="_parent"
455       *    href="http://tools.ietf.org/html/rfc4291#section-2.5.5.1"
456       *    >http://tools.ietf.org/html/rfc4291</a>
457       *
458       * <p>NOTE: This method is different from
459       * {@link Inet6Address#isIPv4CompatibleAddress} in that it more
460       * correctly classifies {@code "::"} and {@code "::1"} as
461       * proper IPv6 addresses (which they are), NOT IPv4 compatible
462       * addresses (which they are generally NOT considered to be).
463       *
464       * @param ip {@link Inet6Address} to be examined for embedded IPv4
465       *           compatible address format
466       * @return {@code true} if the argument is a valid "compat" address
467       */
468      public static boolean isCompatIPv4Address(Inet6Address ip) {
469        if (!ip.isIPv4CompatibleAddress()) {
470          return false;
471        }
472    
473        byte[] bytes = ip.getAddress();
474        if ((bytes[12] == 0) && (bytes[13] == 0) && (bytes[14] == 0)
475                && ((bytes[15] == 0) || (bytes[15] == 1))) {
476          return false;
477        }
478    
479        return true;
480      }

482      /**
483       * Returns the IPv4 address embedded in an IPv4 compatible address.
484       *
485       * @param ip {@link Inet6Address} to be examined for an embedded
486       *           IPv4 address
487       * @return {@link Inet4Address} of the embedded IPv4 address
488       * @throws IllegalArgumentException if the argument is not a valid
489       *         IPv4 compatible address
490       */
491      public static Inet4Address getCompatIPv4Address(Inet6Address ip) {
492        Preconditions.checkArgument(isCompatIPv4Address(ip),
493            "Address '%s' is not IPv4-compatible.", ip.getHostAddress());
494    
495        return getInet4Address(copyOfRange(ip.getAddress(), 12, 16));
496      }

498      /**
499       * Evaluates whether the argument is a 6to4 address.
500       *
501       * <p>6to4 addresses begin with the {@code "2002::/16"} prefix.
502       * The next 32 bits are the IPv4 address of the host to which
503       * IPv6-in-IPv4 tunneled packets should be routed.
504       *
505       * <p>For more on 6to4 addresses see section 2 of
506       * <a target="_parent" href="http://tools.ietf.org/html/rfc3056#section-2"
507       *    >http://tools.ietf.org/html/rfc3056</a>
508       *
509       * @param ip {@link Inet6Address} to be examined for 6to4 address
510       *        format
511       * @return {@code true} if the argument is a 6to4 address
512       */
513      public static boolean is6to4Address(Inet6Address ip) {
514        byte[] bytes = ip.getAddress();
515        return (bytes[0] == (byte) 0x20) && (bytes[1] == (byte) 0x02);
516      }

519       * Returns the IPv4 address embedded in a 6to4 address.
520       *
521       * @param ip {@link Inet6Address} to be examined for embedded IPv4
522       *           in 6to4 address.
523       * @return {@link Inet4Address} of embedded IPv4 in 6to4 address.
524       * @throws IllegalArgumentException if the argument is not a valid
525       *         IPv6 6to4 address.
526       */
527      public static Inet4Address get6to4IPv4Address(Inet6Address ip) {
528        Preconditions.checkArgument(is6to4Address(ip),
529            "Address '%s' is not a 6to4 address.", ip.getHostAddress());
530    
531        return getInet4Address(copyOfRange(ip.getAddress(), 2, 6));
532      }

534      /**
535       * A simple data class to encapsulate the information to be found in a
536       * Teredo address.
537       *
538       * <p>All of the fields in this class are encoded in various portions
539       * of the IPv6 address as part of the protocol.  More protocols details
540       * can be found at:
541       * <a target="_parent" href="http://en.wikipedia.org/wiki/Teredo_tunneling"
542       *    >http://en.wikipedia.org/wiki/Teredo_tunneling</a>.
543       *
544       * <p>The RFC can be found here:
545       * <a target="_parent" href="http://tools.ietf.org/html/rfc4380"
546       *    >http://tools.ietf.org/html/rfc4380</a>.
547       *
548       * @since 5
549       */
550      @Beta
551      public static final class TeredoInfo {
552        private final Inet4Address server;
553        private final Inet4Address client;
554        private final int port;
555        private final int flags;
556    
557        /**
558         * Constructs a TeredoInfo instance.
559         *
560         * <p>Both server and client can be {@code null}, in which case the
561         * value {@code "0.0.0.0"} will be assumed.
562         *
563         * @throws IllegalArgumentException if either of the {@code port}
564         *         or the {@code flags} arguments are out of range of an
565         *         unsigned short
566         */
567        // TODO: why is this public?
568        public TeredoInfo(@Nullable Inet4Address server,
569                          @Nullable Inet4Address client,
570                          int port, int flags) {
571          Preconditions.checkArgument((port >= 0) && (port <= 0xffff),
572              "port '%d' is out of range (0 <= port <= 0xffff)", port);
573          Preconditions.checkArgument((flags >= 0) && (flags <= 0xffff),
574              "flags '%d' is out of range (0 <= flags <= 0xffff)", flags);
575    
576          if (server != null) {
577            this.server = server;
578          } else {
579            this.server = ANY4;
580          }
581    
582          if (client != null) {
583            this.client = client;
584          } else {
585            this.client = ANY4;
586          }
587    
588          this.port = port;
589          this.flags = flags;
590        }
591    
592        public Inet4Address getServer() {
593          return server;
594        }
595    
596        public Inet4Address getClient() {
597          return client;
598        }
599    
600        public int getPort() {
601          return port;
602        }
603    
604        public int getFlags() {
605          return flags;
606        }
607      }

609      /**
610       * Evaluates whether the argument is a Teredo address.
611       *
612       * <p>Teredo addresses begin with the {@code "2001::/32"} prefix.
613       *
614       * @param ip {@link Inet6Address} to be examined for Teredo address
615       *        format.
616       * @return {@code true} if the argument is a Teredo address
617       */
618      public static boolean isTeredoAddress(Inet6Address ip) {
619        byte[] bytes = ip.getAddress();
620        return (bytes[0] == (byte) 0x20) && (bytes[1] == (byte) 0x01)
621               && (bytes[2] == 0) && (bytes[3] == 0);
622      }

625       * Returns the Teredo information embedded in a Teredo address.
626       *
627       * @param ip {@link Inet6Address} to be examined for embedded Teredo
628       *           information
629       * @return extracted {@code TeredoInfo}
630       * @throws IllegalArgumentException if the argument is not a valid
631       *         IPv6 Teredo address
632       */
633      public static TeredoInfo getTeredoInfo(Inet6Address ip) {
634        Preconditions.checkArgument(isTeredoAddress(ip),
635            "Address '%s' is not a Teredo address.", ip.getHostAddress());
636    
637        byte[] bytes = ip.getAddress();
638        Inet4Address server = getInet4Address(copyOfRange(bytes, 4, 8));
639    
640        int flags = ByteStreams.newDataInput(bytes, 8).readShort() & 0xffff;
641    
642        // Teredo obfuscates the mapped client port, per section 4 of the RFC.
643        int port = ~ByteStreams.newDataInput(bytes, 10).readShort() & 0xffff;
644    
645        byte[] clientBytes = copyOfRange(bytes, 12, 16);
646        for (int i = 0; i < clientBytes.length; i++) {
647          // Teredo obfuscates the mapped client IP, per section 4 of the RFC.
648          clientBytes[i] = (byte) ~clientBytes[i];
649        }
650        Inet4Address client = getInet4Address(clientBytes);
651    
652        return new TeredoInfo(server, client, port, flags);
653      }

655      /**
656       * Evaluates whether the argument is an ISATAP address.
657       *
658       * <p>From RFC 5214: "ISATAP interface identifiers are constructed in
659       * Modified EUI-64 format [...] by concatenating the 24-bit IANA OUI
660       * (00-00-5E), the 8-bit hexadecimal value 0xFE, and a 32-bit IPv4
661       * address in network byte order [...]"
662       *
663       * <p>For more on ISATAP addresses see section 6.1 of
664       * <a target="_parent" href="http://tools.ietf.org/html/rfc5214#section-6.1"
665       *    >http://tools.ietf.org/html/rfc5214</a>
666       *
667       * @param ip {@link Inet6Address} to be examined for ISATAP address
668       *        format.
669       * @return {@code true} if the argument is an ISATAP address
670       */
671      public static boolean isIsatapAddress(Inet6Address ip) {
672    
673        // If it's a Teredo address with the right port (41217, or 0xa101)
674        // which would be encoded as 0x5efe then it can't be an ISATAP address.
675        if (isTeredoAddress(ip)) {
676          return false;
677        }
678    
679        byte[] bytes = ip.getAddress();
680    
681        if ((bytes[8] | (byte) 0x03) != (byte) 0x03) {
682    
683          // Verify that high byte of the 64 bit identifier is zero, modulo
684          // the U/L and G bits, with which we are not concerned.
685          return false;
686        }
687    
688        return (bytes[9] == (byte) 0x00) && (bytes[10] == (byte) 0x5e)
689               && (bytes[11] == (byte) 0xfe);
690      }

693       * Returns the IPv4 address embedded in an ISATAP address.
694       *
695       * @param ip {@link Inet6Address} to be examined for embedded IPv4
696       *           in ISATAP address
697       * @return {@link Inet4Address} of embedded IPv4 in an ISATAP address
698       * @throws IllegalArgumentException if the argument is not a valid
699       *         IPv6 ISATAP address
700       */
701      public static Inet4Address getIsatapIPv4Address(Inet6Address ip) {
702        Preconditions.checkArgument(isIsatapAddress(ip),
703            "Address '%s' is not an ISATAP address.", ip.getHostAddress());
704    
705        return getInet4Address(copyOfRange(ip.getAddress(), 12, 16));

708      /**
709       * Examines the Inet6Address to determine if it is an IPv6 address of one
710       * of the specified address types that contain an embedded IPv4 address.
711       *
712       * <p>NOTE: ISATAP addresses are explicitly excluded from this method
713       * due to their trivial spoofability.  With other transition addresses
714       * spoofing involves (at least) infection of one's BGP routing table.
715       *
716       * @param ip {@link Inet6Address} to be examined for embedded IPv4
717       *           client address.
718       * @return {@code true} if there is an embedded IPv4 client address.
719       * @since 7
720       */
721      public static boolean hasEmbeddedIPv4ClientAddress(Inet6Address ip) {
722        return isCompatIPv4Address(ip) || is6to4Address(ip) ||
723               isTeredoAddress(ip);
724      }

726      /**
727       * Examines the Inet6Address to extract the embedded IPv4 client address
728       * if the InetAddress is an IPv6 address of one of the specified address
729       * types that contain an embedded IPv4 address.
730       *
731       * <p>NOTE: ISATAP addresses are explicitly excluded from this method
732       * due to their trivial spoofability.  With other transition addresses
733       * spoofing involves (at least) infection of one's BGP routing table.
734       *
735       * @param ip {@link Inet6Address} to be examined for embedded IPv4
736       *           client address.
737       * @return {@link Inet4Address} of embedded IPv4 client address.
738       * @throws IllegalArgumentException if the argument does not have a valid
739       *         embedded IPv4 address.
740       */
741      public static Inet4Address getEmbeddedIPv4ClientAddress(Inet6Address ip) {
742        if (isCompatIPv4Address(ip)) {
743          return getCompatIPv4Address(ip);
744        }
745    
746        if (is6to4Address(ip)) {
747          return get6to4IPv4Address(ip);
748        }
749    
750        if (isTeredoAddress(ip)) {
751          return getTeredoInfo(ip).getClient();
752        }
753    
754        throw new IllegalArgumentException(
755            String.format("'%s' has no embedded IPv4 address.",
756                          ip.getHostAddress()));
757      }

759      /**
760       * Coerces an IPv6 address into an IPv4 address.
761       *
762       * <p>HACK: As long as applications continue to use IPv4 addresses for
763       * indexing into tables, accounting, et cetera, it may be necessary to
764       * <b>coerce</b> IPv6 addresses into IPv4 addresses. This function does
765       * so by hashing the upper 64 bits into {@code 224.0.0.0/3}
766       * (64 bits into 29 bits).
767       *
768       * <p>A "coerced" IPv4 address is equivalent to itself.
769       *
770       * <p>NOTE: This function is failsafe for security purposes: ALL IPv6
771       * addresses (except localhost (::1)) are hashed to avoid the security
772       * risk associated with extracting an embedded IPv4 address that might
773       * permit elevated privileges.
774       *
775       * @param ip {@link InetAddress} to "coerce"
776       * @return {@link Inet4Address} represented "coerced" address
777       * @since 7
778       */
779      public static Inet4Address getCoercedIPv4Address(InetAddress ip) {
780        if (ip instanceof Inet4Address) {
781          return (Inet4Address) ip;
782        }
783    
784        // Special cases:
785        byte[] bytes = ip.getAddress();
786        boolean leadingBytesOfZero = true;
787        for (int i = 0; i < 15; ++i) {
788          if (bytes[i] != 0) {
789            leadingBytesOfZero = false;
790            break;
791          }
792        }
793        if (leadingBytesOfZero && (bytes[15] == 1)) {
794          return LOOPBACK4;  // ::1
795        } else if (leadingBytesOfZero && (bytes[15] == 0)) {
796          return ANY4;  // ::0
797        }
798    
799        Inet6Address ip6 = (Inet6Address) ip;
800        long addressAsLong = 0;
801        if (hasEmbeddedIPv4ClientAddress(ip6)) {
802          addressAsLong = (long) getEmbeddedIPv4ClientAddress(ip6).hashCode();
803        } else {
804    
805          // Just extract the high 64 bits (assuming the rest is user-modifiable).
806          addressAsLong = ByteBuffer.wrap(ip6.getAddress(), 0, 8).getLong();
807        }
808    
809        // Many strategies for hashing are possible.  This might suffice for now.
810        int coercedHash = Longs.hashCode(addressAsLong);
811    
812        // Squash into 224/4 Multicast and 240/4 Reserved space (i.e. 224/3).
813        coercedHash |= 0xe0000000;
814    
815        // Fixup to avoid some "illegal" values.  Currently the only potential
816        // illegal value is 255.255.255.255.
817        if (coercedHash == 0xffffffff) {
818          coercedHash = 0xfffffffe;
819        }
820    
821        return getInet4Address(Ints.toByteArray(coercedHash));
822      }

824      /**
825       * Returns an integer representing an IPv4 address regardless of
826       * whether the supplied argument is an IPv4 address or not.
827       *
828       * <p>IPv6 addresses are <b>coerced</b> to IPv4 addresses before being
829       * converted to integers.
830       *
831       * <p>As long as there are applications that assume that all IP addresses
832       * are IPv4 addresses and can therefore be converted safely to integers
833       * (for whatever purpose) this function can be used to handle IPv6
834       * addresses as well until the application is suitably fixed.
835       *
836       * <p>NOTE: an IPv6 address coerced to an IPv4 address can only be used
837       * for such purposes as rudimentary identification or indexing into a
838       * collection of real {@link InetAddress}es.  They cannot be used as
839       * real addresses for the purposes of network communication.
840       *
841       * @param ip {@link InetAddress} to convert
842       * @return {@code int}, "coerced" if ip is not an IPv4 address
843       * @since 7
844       */
845      public static int coerceToInteger(InetAddress ip) {
846        return ByteStreams.newDataInput(getCoercedIPv4Address(ip).getAddress()).readInt();
847      }

860      /**
861       * Returns an address from a <b>little-endian ordered</b> byte array
862       * (the opposite of what {@link InetAddress#getByAddress} expects).
863       *
864       * <p>IPv4 address byte array must be 4 bytes long and IPv6 byte array
865       * must be 16 bytes long.
866       *
867       * @param addr the raw IP address in little-endian byte order
868       * @return an InetAddress object created from the raw IP address
869       * @throws UnknownHostException if IP address is of illegal length
870       */
871      public static InetAddress fromLittleEndianByteArray(byte[] addr)
872          throws UnknownHostException {
873        byte[] reversed = new byte[addr.length];
874        for (int i = 0; i < addr.length; i++) {
875          reversed[i] = addr[addr.length - i - 1];
876        }
877        return InetAddress.getByAddress(reversed);
878      }

880      /**
881       * This method emulates the Java 6 method
882       * {@code Arrays.copyOfRange(byte, int, int)}, which is not available in
883       * Java 5, and thus cannot be used in Guava code.
884       */
885      private static byte[] copyOfRange(byte[] original, int from, int to) {
886        Preconditions.checkNotNull(original);
887    
888        int end = Math.min(to, original.length);
889        byte[] result = new byte[to - from];
890    
891        System.arraycopy(original, from, result, 0, end - from);
892        return result;
893      }
