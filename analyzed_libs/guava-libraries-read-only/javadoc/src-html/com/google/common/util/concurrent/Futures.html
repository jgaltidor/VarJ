053      /**
054       * Returns an uninterruptible view of a {@code Future}. If a thread is
055       * interrupted during an attempt to {@code get()} from the returned future, it
056       * continues to wait on the result until it is available or the timeout
057       * elapses, and only then re-interrupts the thread.
058       */
059      public static <V> UninterruptibleFuture<V> makeUninterruptible(
060          final Future<V> future) {
061        checkNotNull(future);
062        if (future instanceof UninterruptibleFuture<?>) {
063          return (UninterruptibleFuture<V>) future;
064        }
065        return new UninterruptibleFuture<V>() {
066          public boolean cancel(boolean mayInterruptIfRunning) {
067            return future.cancel(mayInterruptIfRunning);
068          }
069          public boolean isCancelled() {
070            return future.isCancelled();
071          }
072          public boolean isDone() {
073            return future.isDone();
074          }
075    
076          public V get(long timeoutDuration, TimeUnit timeoutUnit)
077              throws TimeoutException, ExecutionException {
078            boolean interrupted = false;
079            try {
080              long timeoutNanos = timeoutUnit.toNanos(timeoutDuration);
081              long end = System.nanoTime() + timeoutNanos;
082              while (true) {
083                try {
084                  return future.get(timeoutNanos, NANOSECONDS);
085                } catch (InterruptedException e) {
086                  // Future treats negative timeouts just like zero.
087                  timeoutNanos = end - System.nanoTime();
088                  interrupted = true;
089                }
090              }
091            } finally {
092              if (interrupted) {
093                Thread.currentThread().interrupt();
094              }
095            }
096          }
097    
098          public V get() throws ExecutionException {
099            boolean interrupted = false;
100            try {
101              while (true) {
102                try {
103                  return future.get();
104                } catch (InterruptedException ignored) {
105                  interrupted = true;
106                }
107              }
108            } finally {
109              if (interrupted) {
110                Thread.currentThread().interrupt();
111              }
112            }
113          }
114        };
115      }

170       * Creates a {@code ListenableFuture} which has its value set immediately upon
171       * construction. The getters just return the value. This {@code Future} can't
172       * be canceled or timed out and its {@code isDone()} method always returns
173       * {@code true}. It's useful for returning something that implements the
174       * {@code ListenableFuture} interface but already has the result.
175       */
176      public static <V> ListenableFuture<V> immediateFuture(@Nullable V value) {
177        ValueFuture<V> future = ValueFuture.create();
178        future.set(value);
179        return future;

183       * Returns a {@code CheckedFuture} which has its value set immediately upon
184       * construction.
185       *
186       * <p>The returned {@code Future} can't be cancelled, and its {@code isDone()}
187       * method always returns {@code true}. Calling {@code get()} or {@code
188       * checkedGet()} will immediately return the provided value.

190      public static <V, X extends Exception> CheckedFuture<V, X>

192        ValueFuture<V> future = ValueFuture.create();
193        future.set(value);
194        return Futures.makeChecked(future, new Function<Exception, X>() {
195          public X apply(Exception e) {
196            throw new AssertionError("impossible");

243      /**
244       * Returns a new {@code ListenableFuture} whose result is asynchronously
245       * derived from the result of the given {@code Future}. More precisely, the
246       * returned {@code Future} takes its result from a {@code Future} produced by
247       * applying the given {@code Function} to the result of the original {@code
248       * Future}.
249       *
250       * <p>Successful cancellation of either the input future or the result of
251       * function application will cause the returned future to be cancelled.
252       * Cancelling the returned future will succeed if it is currently running.
253       * In this case, attempts will be made to cancel the input future and the
254       * result of the function, however there is no guarantee of success.
255       *
256       * <p>TODO: Add a version that accepts a normal {@code Future}
257       *
258       * <p>The typical use for this method would be when a RPC call is dependent on
259       * the results of another RPC.  One would call the first RPC (input), create a
260       * function that calls another RPC based on input's result, and then call
261       * chain on input and that function to get a {@code ListenableFuture} of
262       * the result.
263       *
264       * @param input The future to chain
265       * @param function A function to chain the results of the provided future
266       *     to the results of the returned future.  This will be run in the thread
267       *     that notifies input it is complete.
268       * @return A future that holds result of the chain.
269       */
270      public static <I, O> ListenableFuture<O> chain(ListenableFuture<I> input,
271          Function<? super I, ? extends ListenableFuture<? extends O>> function) {
272        return chain(input, function, MoreExecutors.sameThreadExecutor());
273      }

276       * Returns a new {@code ListenableFuture} whose result is asynchronously
277       * derived from the result of the given {@code Future}. More precisely, the
278       * returned {@code Future} takes its result from a {@code Future} produced by
279       * applying the given {@code Function} to the result of the original {@code
280       * Future}.
281       *
282       * <p>Successful cancellation of either the input future or the result of
283       * function application will cause the returned future to be cancelled.
284       * Cancelling the returned future will succeed if it is currently running.
285       * In this case, attempts will be made to cancel the input future and the
286       * result of the function, however there is no guarantee of success.
287       *
288       * <p>This version allows an arbitrary executor to be passed in for running
289       * the chained Function. When using {@link MoreExecutors#sameThreadExecutor},
290       * the thread chained Function executes in will be whichever thread set the
291       * result of the input Future, which may be the network thread in the case of
292       * RPC-based Futures.
293       *
294       * @param input The future to chain
295       * @param function A function to chain the results of the provided future
296       *     to the results of the returned future.
297       * @param exec Executor to run the function in.
298       * @return A future that holds result of the chain.
299       */
300      public static <I, O> ListenableFuture<O> chain(ListenableFuture<I> input,
301          Function<? super I, ? extends ListenableFuture<? extends O>> function,
302          Executor exec) {
303        ChainingListenableFuture<I, O> chain =
304            new ChainingListenableFuture<I, O>(function, input);
305        input.addListener(chain, exec);
306        return chain;
307      }

333      /**
334       * Returns a new {@code ListenableFuture} whose result is the product of
335       * applying the given {@code Function} to the result of the given {@code
336       * Future}.
337       *
338       * <p>Successful cancellation of the input future will cause the returned
339       * future to be cancelled.  Cancelling the returned future will succeed if it
340       * is currently running.  In this case, an attempt will be made to cancel the
341       * input future, however there is no guarantee of success.
342       *
343       * <p>An example use of this method is to convert a serializable object
344       * returned from an RPC into a POJO.
345       *
346       * <p>This version allows an arbitrary executor to be passed in for running
347       * the chained Function. When using {@link MoreExecutors#sameThreadExecutor},
348       * the thread chained Function executes in will be whichever thread set the
349       * result of the input Future, which may be the network thread in the case of
350       * RPC-based Futures.
351       *
352       * @param future The future to compose
353       * @param function A Function to compose the results of the provided future
354       *     to the results of the returned future.
355       * @param exec Executor to run the function in.
356       * @return A future that holds result of the composition.
357       * @since 2
358       */
359      public static <I, O> ListenableFuture<O> compose(ListenableFuture<I> future,
360          final Function<? super I, ? extends O> function, Executor exec) {
361        checkNotNull(function);
362        Function<I, ListenableFuture<O>> wrapperFunction
363            = new Function<I, ListenableFuture<O>>() {
364                @Override public ListenableFuture<O> apply(I input) {
365                  O output = function.apply(input);
366                  return immediateFuture(output);
367                }
368            };
369        return chain(future, wrapperFunction, exec);
370      }

372      /**
373       * Returns a new {@code Future} whose result is the product of applying the
374       * given {@code Function} to the result of the given {@code Future}.
375       *
376       * <p>An example use of this method is to convert a Future that produces a
377       * handle to an object to a future that produces the object itself.
378       *
379       * <p>Each call to {@code Future<O>.get(*)} results in a call to
380       * {@code Future<I>.get(*)}, but {@code function} is only applied once, so it
381       * is assumed that {@code Future<I>.get(*)} is idempotent.
382       *
383       * <p>When calling {@link Future#get(long, TimeUnit)} on the returned
384       * future, the timeout only applies to the future passed in to this method.
385       * Any additional time taken by applying {@code function} is not considered.
386       * (Exception: If the input future is a {@link ListenableFuture}, timeouts
387       * will be strictly enforced.)
388       *
389       * @param future The future to compose
390       * @param function A Function to compose the results of the provided future
391       *     to the results of the returned future.  This will be run in the thread
392       *     that calls one of the varieties of {@code get()}.
393       * @return A future that computes result of the composition.
394       */
395      public static <I, O> Future<O> compose(final Future<I> future,
396          final Function<? super I, ? extends O> function) {
397        if (future instanceof ListenableFuture) {
398          return compose((ListenableFuture<I>) future, function);
399        }
400        checkNotNull(future);
401        checkNotNull(function);
402        return new Future<O>() {
403    
404          /*
405           * Concurrency detail:
406           *
407           * <p>To preserve the idempotency of calls to this.get(*) calls to the
408           * function are only applied once. A lock is required to prevent multiple
409           * applications of the function. The calls to future.get(*) are performed
410           * outside the lock, as is required to prevent calls to
411           * get(long, TimeUnit) to persist beyond their timeout.
412           *
413           * <p>Calls to future.get(*) on every call to this.get(*) also provide
414           * the cancellation behavior for this.
415           *
416           * <p>(Consider: in thread A, call get(), in thread B call get(long,
417           * TimeUnit). Thread B may have to wait for Thread A to finish, which
418           * would be unacceptable.)
419           *
420           * <p>Note that each call to Future<O>.get(*) results in a call to
421           * Future<I>.get(*), but the function is only applied once, so
422           * Future<I>.get(*) is assumed to be idempotent.
423           */
424    
425          private final Object lock = new Object();
426          private boolean set = false;
427          private O value = null;
428          private ExecutionException exception = null;
429    
430          @Override
431          public O get() throws InterruptedException, ExecutionException {
432            return apply(future.get());
433          }
434    
435          @Override
436          public O get(long timeout, TimeUnit unit) throws InterruptedException,
437              ExecutionException, TimeoutException {
438            return apply(future.get(timeout, unit));
439          }
440    
441          private O apply(I raw) throws ExecutionException {
442            synchronized (lock) {
443              if (!set) {
444                try {
445                  value = function.apply(raw);
446                } catch (RuntimeException e) {
447                  exception = new ExecutionException(e);
448                } catch (Error e) {
449                  exception = new ExecutionException(e);
450                }
451                set = true;
452              }
453    
454              if (exception != null) {
455                throw exception;
456              }
457              return value;
458            }
459          }
460    
461          @Override
462          public boolean cancel(boolean mayInterruptIfRunning) {
463            return future.cancel(mayInterruptIfRunning);
464          }
465    
466          @Override
467          public boolean isCancelled() {
468            return future.isCancelled();
469          }
470    
471          @Override
472          public boolean isDone() {
473            return future.isDone();
474          }
475        };
476      }

478      /**
479       * An implementation of {@code ListenableFuture} that also implements
480       * {@code Runnable} so that it can be used to nest ListenableFutures.
481       * Once the passed-in {@code ListenableFuture} is complete, it calls the
482       * passed-in {@code Function} to generate the result.
483       *
484       * <p>If the function throws any checked exceptions, they should be wrapped
485       * in a {@code UndeclaredThrowableException} so that this class can get
486       * access to the cause.
487       */
488      private static class ChainingListenableFuture<I, O>
489          extends AbstractListenableFuture<O> implements Runnable {
490    
491        private Function<? super I, ? extends ListenableFuture<? extends O>>
492            function;
493        private ListenableFuture<? extends I> inputFuture;
494        private volatile ListenableFuture<? extends O> outputFuture;
495        private final BlockingQueue<Boolean> mayInterruptIfRunningChannel =
496            new LinkedBlockingQueue<Boolean>(1);
497        private final CountDownLatch outputCreated = new CountDownLatch(1);
498    
499        private ChainingListenableFuture(
500            Function<? super I, ? extends ListenableFuture<? extends O>> function,
501            ListenableFuture<? extends I> inputFuture) {
502          this.function = checkNotNull(function);
503          this.inputFuture = checkNotNull(inputFuture);
504        }
505    
506        /**
507         * Delegate the get() to the input and output futures, in case
508         * their implementations defer starting computation until their
509         * own get() is invoked.
510         */
511        public O get() throws InterruptedException, ExecutionException {
512          if (!isDone()) {
513            // Invoking get on the inputFuture will ensure our own run()
514            // method below is invoked as a listener when inputFuture sets
515            // its value.  Therefore when get() returns we should then see
516            // the outputFuture be created.
517            ListenableFuture<? extends I> inputFuture = this.inputFuture;
518            if (inputFuture != null) {
519              inputFuture.get();
520            }
521    
522            // If our listener was scheduled to run on an executor we may
523            // need to wait for our listener to finish running before the
524            // outputFuture has been constructed by the function.
525            outputCreated.await();
526    
527            // Like above with the inputFuture, we have a listener on
528            // the outputFuture that will set our own value when its
529            // value is set.  Invoking get will ensure the output can
530            // complete and invoke our listener, so that we can later
531            // get the result.
532            ListenableFuture<? extends O> outputFuture = this.outputFuture;
533            if (outputFuture != null) {
534              outputFuture.get();
535            }
536          }
537          return super.get();
538        }
539    
540        /**
541         * Delegate the get() to the input and output futures, in case
542         * their implementations defer starting computation until their
543         * own get() is invoked.
544         */
545        public O get(long timeout, TimeUnit unit) throws TimeoutException,
546            ExecutionException, InterruptedException {
547          if (!isDone()) {
548            // Use a single time unit so we can decrease remaining timeout
549            // as we wait for various phases to complete.
550            if (unit != NANOSECONDS) {
551              timeout = NANOSECONDS.convert(timeout, unit);
552              unit = NANOSECONDS;
553            }
554    
555            // Invoking get on the inputFuture will ensure our own run()
556            // method below is invoked as a listener when inputFuture sets
557            // its value.  Therefore when get() returns we should then see
558            // the outputFuture be created.
559            ListenableFuture<? extends I> inputFuture = this.inputFuture;
560            if (inputFuture != null) {
561              long start = System.nanoTime();
562              inputFuture.get(timeout, unit);
563              timeout -= Math.max(0, System.nanoTime() - start);
564            }
565    
566            // If our listener was scheduled to run on an executor we may
567            // need to wait for our listener to finish running before the
568            // outputFuture has been constructed by the function.
569            long start = System.nanoTime();
570            if (!outputCreated.await(timeout, unit)) {
571              throw new TimeoutException();
572            }
573            timeout -= Math.max(0, System.nanoTime() - start);
574    
575            // Like above with the inputFuture, we have a listener on
576            // the outputFuture that will set our own value when its
577            // value is set.  Invoking get will ensure the output can
578            // complete and invoke our listener, so that we can later
579            // get the result.
580            ListenableFuture<? extends O> outputFuture = this.outputFuture;
581            if (outputFuture != null) {
582              outputFuture.get(timeout, unit);
583            }
584          }
585          return super.get(timeout, unit);
586        }
587    
588        @Override
589        public boolean cancel(boolean mayInterruptIfRunning) {
590          if (cancel()) {
591            try {
592              // This should never block since only one thread is allowed to cancel
593              // this Future.
594              mayInterruptIfRunningChannel.put(mayInterruptIfRunning);
595            } catch (InterruptedException ignored) {
596              Thread.currentThread().interrupt();
597            }
598            cancel(inputFuture, mayInterruptIfRunning);
599            cancel(outputFuture, mayInterruptIfRunning);
600            return true;
601          }
602          return false;
603        }
604    
605        private void cancel(@Nullable Future<?> future,
606            boolean mayInterruptIfRunning) {
607          if (future != null) {
608            future.cancel(mayInterruptIfRunning);
609          }
610        }
611    
612        public void run() {
613          try {
614            I sourceResult;
615            try {
616              sourceResult = makeUninterruptible(inputFuture).get();
617            } catch (CancellationException e) {
618              // Cancel this future and return.
619              cancel();
620              return;
621            } catch (ExecutionException e) {
622              // Set the cause of the exception as this future's exception
623              setException(e.getCause());
624              return;
625            }
626    
627            final ListenableFuture<? extends O> outputFuture = this.outputFuture =
628                function.apply(sourceResult);
629            if (isCancelled()) {
630              // Handles the case where cancel was called while the function was
631              // being applied.
632              try {
633                // There is a gap in cancel(boolean) between calling cancel() and
634                // storing the value of mayInterruptIfRunning, so this thread needs
635                // to block, waiting for that value.
636                outputFuture.cancel(mayInterruptIfRunningChannel.take());
637              } catch (InterruptedException ignored) {
638                Thread.currentThread().interrupt();
639              }
640              this.outputFuture = null;
641              return;
642            }
643            outputFuture.addListener(new Runnable() {
644                public void run() {
645                  try {
646                    // Here it would have been nice to have had an
647                    // UninterruptibleListenableFuture, but we don't want to start a
648                    // combinatorial explosion of interfaces, so we have to make do.
649                    set(makeUninterruptible(outputFuture).get());
650                  } catch (CancellationException e) {
651                    // Cancel this future and return.
652                    cancel();
653                    return;
654                  } catch (ExecutionException e) {
655                    // Set the cause of the exception as this future's exception
656                    setException(e.getCause());
657                  } finally {
658                    // Don't pin inputs beyond completion
659                    ChainingListenableFuture.this.outputFuture = null;
660                  }
661                }
662              }, MoreExecutors.sameThreadExecutor());
663          } catch (UndeclaredThrowableException e) {
664            // Set the cause of the exception as this future's exception
665            setException(e.getCause());
666          } catch (RuntimeException e) {
667            // This exception is irrelevant in this thread, but useful for the
668            // client
669            setException(e);
670          } catch (Error e) {
671            // Propagate errors up ASAP - our superclass will rethrow the error
672            setException(e);
673          } finally {
674            // Don't pin inputs beyond completion
675            function = null;
676            inputFuture = null;
677            // Allow our get routines to examine outputFuture now.
678            outputCreated.countDown();
679          }
680        }
681      }
682    
683      /**
684       * A checked future that uses a function to map from exceptions to the
685       * appropriate checked type.
686       */
687      private static class MappingCheckedFuture<V, X extends Exception> extends
688          AbstractCheckedFuture<V, X> {
689    
690        final Function<Exception, X> mapper;
691    
692        MappingCheckedFuture(ListenableFuture<V> delegate,
693            Function<Exception, X> mapper) {
694          super(delegate);
695    
696          this.mapper = checkNotNull(mapper);
697        }
698    
699        @Override
700        protected X mapException(Exception e) {
701          return mapper.apply(e);
702        }
703      }

705      /**
706       * An adapter to turn a {@link Future} into a {@link ListenableFuture}.  This
707       * will wait on the future to finish, and when it completes, run the
708       * listeners.  This implementation will wait on the source future
709       * indefinitely, so if the source future never completes, the adapter will
710       * never complete either.
711       *
712       * <p>If the delegate future is interrupted or throws an unexpected unchecked
713       * exception, the listeners will not be invoked.
714       */
715      private static class ListenableFutureAdapter<V> extends ForwardingFuture<V>
716          implements ListenableFuture<V> {
717    
718        private static final ThreadFactory threadFactory =
719            new ThreadFactoryBuilder()
720                .setNameFormat("ListenableFutureAdapter-thread-%d")
721                .build();
722        private static final Executor defaultAdapterExecutor =
723            Executors.newCachedThreadPool(threadFactory);
724    
725        private final Executor adapterExecutor;
726    
727        // The execution list to hold our listeners.
728        private final ExecutionList executionList = new ExecutionList();
729    
730        // This allows us to only start up a thread waiting on the delegate future
731        // when the first listener is added.
732        private final AtomicBoolean hasListeners = new AtomicBoolean(false);
733    
734        // The delegate future.
735        private final Future<V> delegate;
736    
737        ListenableFutureAdapter(Future<V> delegate) {
738          this(delegate, defaultAdapterExecutor);
739        }
740    
741        ListenableFutureAdapter(Future<V> delegate, Executor adapterExecutor) {
742          this.delegate = checkNotNull(delegate);
743          this.adapterExecutor = checkNotNull(adapterExecutor);
744        }
745    
746        @Override
747        protected Future<V> delegate() {
748          return delegate;
749        }
750    
751        @Override
752        public void addListener(Runnable listener, Executor exec) {
753          executionList.add(listener, exec);
754    
755          // When a listener is first added, we run a task that will wait for
756          // the delegate to finish, and when it is done will run the listeners.
757          if (!hasListeners.get() && hasListeners.compareAndSet(false, true)) {
758            if (delegate.isDone()) {
759              // If the delegate is already done, run the execution list
760              // immediately on the current thread.
761              executionList.run();
762              return;
763            }
764    
765            adapterExecutor.execute(new Runnable() {
766              @Override
767              public void run() {
768                try {
769                  delegate.get();
770                } catch (CancellationException e) {
771                  // The task was cancelled, so it is done, run the listeners.
772                } catch (InterruptedException e) {
773                  // This thread was interrupted.  This should never happen, so we
774                  // throw an IllegalStateException.
775                  Thread.currentThread().interrupt();
776                  throw new IllegalStateException(
777                      "Adapter thread interrupted!", e);
778                } catch (ExecutionException e) {
779                  // The task caused an exception, so it is done, run the
780                  // listeners.
781                }
782                executionList.run();
783              }
784            });
785          }
786        }
787      }
